whats-element.js

 为每一个 HTML 元素都很特别

将鼠标移至任意元素上， 双击或右击即可计算得出该HTML元素的唯一标识符

Download whatsElement.js or View on GitHub

example

      
        <script src="../dist/whatsElement.js"></script >
        <script>
           var whats = new whatsElement();
           document.addEventListener('dblclick',function (event) {
             var result = whats.getUniqueId(event.target);
             var elementId = result.uniqueId;// 该元素的唯一标识符
             var queryType = result.queryType;// 定位该元素的类型，如 id,class,name...
           })
       </script>
